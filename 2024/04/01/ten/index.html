  1. 最长公共前缀
    简单
    相关标签
    相关企业
    编写一个函数来查找字符串数组中的最长公共前缀。

如果不存在公共前缀，返回空字符串 “”。

 

示例 1：

输入：strs = [“flower”,”flow”,”flight”]
输出：”fl”
示例 2：

输入：strs = [“dog”,”racecar”,”car”]
输出：””
解释：输入不存在公共前缀。  

提示：

1 <= strs.length <= 200
0 <= strs[i].length <= 200
strs[i] 仅由小写英文字母组成

/**

  • @param {string[]} strs - 输入参数，一个字符串数组

  • @return {string} - 返回最长公共前缀
    */
    var longestCommonPrefix = function(strs) {
    // 如果输入的字符串数组为空，则最长公共前缀为空字符串
    if(strs.length == 0)
    return “”;
    // 初始化最长公共前缀为数组中的第一个字符串
    let ans = strs[0];

    // 遍历数组中的每一个字符串
    for(let i = 1; i < strs.length; i++) {
    let j = 0;

    // 在当前字符串和最长公共前缀之间进行字符比较
    for(; j < ans.length && j < strs[i].length; j++) {
    // 如果当前字符不匹配，则跳出循环
    if(ans[j] != strs[i][j])
    break;
    }

    // 更新最长公共前缀为当前已比较的部分
    ans = ans.substr(0, j);

    // 如果最长公共前缀为空，则直接返回空字符串
    if(ans === “”)
    return ans;
    }

    // 返回最终的最长公共前缀
    return ans;

};

  1. 三数之和
    中等
    相关标签
    相关企业
    提示
    给你一个整数数组 nums ，判断是否存在三元组 [nums[i], nums[j], nums[k]] 满足 i != j、i != k 且 j != k ，同时还满足 nums[i] + nums[j] + nums[k] == 0 。请

你返回所有和为 0 且不重复的三元组。

注意：答案中不可以包含重复的三元组。

 

 

示例 1：

输入：nums = [-1,0,1,2,-1,-4]
输出：[[-1,-1,2],[-1,0,1]]
解释：
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0 。
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0 。
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0 。
不同的三元组是 [-1,0,1] 和 [-1,-1,2] 。
注意，输出的顺序和三元组的顺序并不重要。
示例 2：

输入：nums = [0,1,1]
输出：[]
解释：唯一可能的三元组和不为 0 。
示例 3：

输入：nums = [0,0,0]
输出：[[0,0,0]]
解释：唯一可能的三元组和为 0 。

/**

  • @param {number[]} nums - 输入参数，一个整数数组

  • @return {number[][]} - 返回三数之和的所有可能组合
    */
    var threeSum = function(nums) {
    let ans = []; // 用于存储结果数组
    const len = nums.length; // 获取数组长度
    if(nums == null || len < 3) return ans; // 如果数组为空或长度小于3，直接返回空数组

    // 对数组进行升序排序
    nums.sort((a, b) => a - b);

    // 遍历数组
    for (let i = 0; i < len ; i++) {
    // 如果当前数字大于0，则三数之和一定大于0，所以结束循环
    if(nums[i] > 0) break;

    // 如果当前数字和前一个数字相同，则跳过此次循环，以避免重复解
    if(i > 0 && nums[i] == nums[i-1]) continue;

    let L = i + 1; // 左指针，初始化为当前元素的下一个位置
    let R = len - 1; // 右指针，初始化为数组最后一个元素的位置

    // 当左指针小于右指针时，进行循环
    while(L < R){
    const sum = nums[i] + nums[L] + nums[R]; // 计算三数之和

    // 如果和为0，则找到了一组解
    if(sum == 0){
    ans.push([nums[i],nums[L],nums[R]]); // 将解添加到结果数组中

    // 跳过重复元素，避免重复解
    while (L < R && nums[L] == nums[L + 1]) L++;
    while (L < R && nums[R] == nums[R - 1]) R–;

    // 移动左右指针，继续寻找新的解
    L++;
    R–;
    }
    // 如果和小于0，则左指针右移，增加和的值
    else if (sum < 0) L++;
    // 如果和大于0，则右指针左移，减小和的值
    else if (sum > 0) R–;
    }
    }

    // 返回所有解
    return ans;

};
